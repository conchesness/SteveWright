Auto Generate Charts!

This page uses the charts.js library.

First you need to add this to your scripts which are probably at the bottom of the html page before the closing body tag.


      <script src="https://cdn.jsdelivr.net/npm/chart.js@3.5.1/dist/chart.min.js"></script>

Then you need to create a "canvas" for the image. notice the 'id=pie'.


      <canvas id="pie" width="400" height="400"></canvas>

Now you need the javascript!


      <script type="text/javascript">

        // This line references the 'id' where the chart will go in the page.
        var pie = document.getElementById('pie').getContext('2d');

        // This is a 'dictionary' of values that define the configuration of the chart.
        // You can find out how to do this for several charts at 
        // https://www.chartjs.org/docs/latest/charts/doughnut.html
        const pieConfig = {
          type: 'doughnut',
          data: {
            labels: [
                'Wright',
                'Nobel',
                'Ketcham'
            ],
            datasets: [{
                // if you have the data in variables you can put those variables here instead of 
                // typing the literal values. This data could come from a form like on the forms 
                // page or it could come out of a database!
                data: [300, 50, 100],
                backgroundColor: [
                  'rgb(255, 99, 132)',
                  'rgb(54, 162, 235)',
                  'rgb(255, 205, 86)'
                ],
                hoverOffset: 4
            }]
          }
        };
      
        // This creates the chart
        var pieChart = new Chart(pie, pieConfig);
      </script>

Here is the sample code for the bar chart


    <script type="text/javascript">
    var bars = document.getElementById('bars').getContext('2d');
    var barConfig = {
        type: 'bar',
        data: {
          labels: ['Red', 'Blue', 'Yellow', 'Green', 'Purple', 'Orange'],
          datasets: [{
            label: '# of Votes',
            data: [12, 19, 3, 5, 2, 3],
            backgroundColor: [
                'rgba(255, 99, 132, 0.2)',
                'rgba(54, 162, 235, 0.2)',
                'rgba(255, 206, 86, 0.2)',
                'rgba(75, 192, 192, 0.2)',
                'rgba(153, 102, 255, 0.2)',
                'rgba(255, 159, 64, 0.2)'
            ],
            borderColor: [
                'rgba(255, 99, 132, 1)',
                'rgba(54, 162, 235, 1)',
                'rgba(255, 206, 86, 1)',
                'rgba(75, 192, 192, 1)',
                'rgba(153, 102, 255, 1)',
                'rgba(255, 159, 64, 1)'
            ],
            borderWidth: 1
          }]
        },
        options: {
            scales: {
                y: {
                    beginAtZero: true
                }
            }
          }
      };
  
      var Chart = new Chart(bars, barConfig);
